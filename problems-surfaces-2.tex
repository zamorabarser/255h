\section{Surface integrals of vector fields}

\begin{exercise}  
    Let  $C\subset \mathbb{R}^3$ be the curve with parametrization $\gamma : [0,1 ] \to \mathbb{R}^3$ given by
    \[      \gamma (t) = (  t + t^2 ,  t ^4 ,  2t + t^2 - 3  ).                      \]
    Compute
    \[    \int_C \langle  - y , x + 2 , z + 1  \rangle \cdot ds      \]
\end{exercise}
