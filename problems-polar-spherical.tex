\begin{abstract}
    Polar, cylindrical, and spherical coordinates
\end{abstract}


\begin{exercise}
    Using polar coordinates, compute
    \[     \int_{-4}^0 \int _{- \sqrt{16 - x^2}} ^0 y \, dy dx     \]
\end{exercise}


\begin{exercise}
    Using polar coordinates, compute
    \[     \int_{0}^{3/ \sqrt{2}}  \int _{ y } ^{\sqrt{9 - y^2 }}  x^2y  \, dx dy     \]
\end{exercise}


\begin{exercise}
    Using polar coordinates, compute
    \[   \iint _B  x  \, dx dy     \]
    where $B\subset \mathbb{R}^2$ is the region in the first quadrant inside the circle $x^2 + (y-3)^2 = 9$
\end{exercise}

\begin{exercise}
    Using polar coordinates, show that
    \[   \int _{- \infty } ^{\infty}   e^{- \frac{x^2}{2} }  \, dx  = \sqrt{2 \pi }     \]
\end{exercise}


\begin{exercise}
    Let $D \subset \mathbb{R}^2$ be the interior of the  circle $(x-1)^2 + (y-1)^2 = 2$. Using polar coordinates, find 
    \[     \iint_D \frac{x}{x^2 + y^2 } dA   \]
    \begin{hint}
        The notes have a section on how to describe these regions in polar coordinates.        
    \end{hint}
\end{exercise}




\begin{exercise}
    Using cylindrical coordinates, compute
    \[   \int _0 ^3  \int_ {- \sqrt{9 - y^2} } ^{\sqrt{9 - y^2}}  \int_ y ^{4}  x ^2 \, dz dxdy  \]
\end{exercise}




\begin{exercise}
    Using cylindrical coordinates, compute
    \[   \int _0 ^ 1   \int_ {- 1 } ^{1}  \int_z ^{ \sqrt{1 - y ^2 }}   y^2  \, dx dy dz  \]
\end{exercise}

\begin{exercise}
    Using spherical coordinates, compute
    \[   \int _{-2} ^2  \int_ { - \sqrt{4 - x^2 } } ^{\sqrt{4 - x^2}}  \int_ {0} ^{\sqrt{4 - x^2 - y^2 }}  y^2  \, dz dydx  \]
\end{exercise}




\begin{exercise}
    Using spherical coordinates, compute
    \[   \iiint _ E   zy^2  \, dV  \]
    where $E$ is the region above the cone $z = -  \sqrt{x^2 + y^2 }$ and inside the sphere $x^2 + y^2 + z^2  = 25 $
\end{exercise}


\begin{exercise}
    Using spherical coordinates, compute
    \[   \iiint _ E  \frac{ z x^2}{x^2 + y^2}  \, dV  \]
    where $E$ is the region below the plane $z = 3$ and above the cone $z = \sqrt{x^2 + y^2} / \sqrt{3}$
\end{exercise}


\begin{exercise}
    Using spherical coordinates, compute
    \[   \iiint _ E  z  \, dV  \]
    where $E$ is the region inside the sphere $x^2 + y^2 + (z-3)^2 = 9 $
\end{exercise}
